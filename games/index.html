// Logic for games/index.html async function loadGames() { const response = await fetch('../assets/games.json'); const games = await response.json(); const container = document.getElementById('games-grid'); const favorites = JSON.parse(localStorage.getItem('favGames') || '[]'); function render(filterTag = '', searchQuery = '') { container.innerHTML = ''; // Sort: Favorites first const sortedGames = [...games].sort((a, b) => { return favorites.includes(b.title) - favorites.includes(a.title); }); sortedGames.forEach(game => { if (filterTag && !game.tags.includes(filterTag)) return; if (searchQuery && !game.title.toLowerCase().includes(searchQuery.toLowerCase())) return; const isFav = favorites.includes(game.title); const card = `

${game.title}

${game.tags.map(t => `${t}`).join('')}
Play
`; container.innerHTML += card; }); } // ... render logic ... }
